{Bourbon Manhattan}

    \ingredients
    {
		2 & oz bourbon \\
		1 & oz sweet vermouth \\
		2 & dashes Angostura bitters \\
		1 & dash orange bitters \\
    }
    
    \preparation
    {
        \step Combine ingredients in mixing glass with ice, and stir until well-chilled. 
		\step Strain and pour into glass, either chilled or over ice. \\
    }


\end{recipe}
