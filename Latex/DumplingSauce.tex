{Dumpling Sauce}

    \ingredients
    {
		1 & tsp sugar \\
		1 & Tbsp hot water \\
		 & \\
		2 & Tbsp soy sauce \\
		1 & tsp rice vinegar \\
		1 & tsp chili oil \\
		1 & tsp garlic (minced) \\
		1 & tsp toasted sesame seeds \\
		$\sfrac{1}{2}$ & tsp sesame oil \\
    }
    
    \preparation
    {
        \step Dissolve sugar in hot water.  \\
		\step Add remaining ingredients and stir to combine. \\
    }
