{1855 Champagne Cocktail}

    \ingredients
    {
		2 & tsp simple syrup \\
		dashes & angostura bitters \\
		 & \\
		2-4 & oz Asti \\
    }
    
    \preparation
    {
        \step Combine simple syrup and angostura bitters in a glass filled with ice and stir. 
		\step Pour in Asti and stir. Strain into old-fashioned glass. 
    }


\end{recipe}
