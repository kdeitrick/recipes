{Red Eye Martini}

    \ingredients
    {
		$\sfrac{1}{2}$ & oz Kahlua \\
		$\sfrac{1}{2}$ & oz Bailey's \\
		$\sfrac{1}{2}$ & oz Frangelico \\
		$\sfrac{3}{4}$ & oz espresso \\
		2 & oz espresso vodka
    }
    
    \preparation
    {
        \step Combine ingredients in cocktail shaker with ice, and shake. 
		\step Strain and pour into glass(es). \\
    }
