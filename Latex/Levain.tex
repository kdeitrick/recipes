\begin{recipe}
[ % Optionale Eingaben
%    preparationtime = {1]{h}},
%    portion = \portion{5},
    source = Zingerman's via Jim,
]
{Zingerman's Levain}

    \ingredients
    {
		102.5 & g active starter \\
		205 & g AP flour \\
		102.5 & g water, room temp \\
    }
    
    \preparation
    {
        \step Take all ingredients and stir to combine. \\
		\step Leave in a covered bowl, for 6-8 hours, before using in the recipe calling for it. \\
    }

	\suggestion[Folding]
	{
		Shape the dough into a rectangle. Fold $\sfrac{1}{3}$ of the dough toward the center and then fold $\sfrac{1}{3}$ of the dough from the other side over that. Repeat vertically, and place the ``package'' of dough seam side down into the proofing container. 
	}
	
	\suggestion[Touch test]
	{
		Poke the bread with a finger. The loaf is ready when the indentation pushes back very slowly and doesn't completely fill back in. If it immediately pushes back out, check again in 15 minutes; if it doesn't spring back at all, bake immediately and score very lightly. 
	}
	
	\suggestion[Baking with steam]
	{
		Place ice cubes into preheated cast iron skillet, place loaves onto baking stone, and cover loaves - making sure cover extends over edge of baking stone, allowing it to capture steam produce by ice cubes. 
	}
	
	\suggestion[Freezing]
	{
		Wrap fresh, cooled loaves in saran wrap, then aluminum foil, before placing in freezer. To thaw, preheat oven to 425$^{\circ}$F. Unwrap and place on baking sheet; bake for $\sim$15 minutes.
	}

\end{recipe}
